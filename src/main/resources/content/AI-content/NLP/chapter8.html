        """

        """

        label_ids = []

            'input_ids': torch.tensor(input_ids),
            'attention_mask': torch.tensor(attention_mask),

    tp, fp, fn = 0, 0, 0

        tp += len(pred_set & label_set)

        fn += len(label_set - pred_set)

    precision = tp / (tp + fp) if tp + fp >0 else 0
    recall = tp / (tp + fn) if tp + fn > 0 else 0

        'precision': precision,
