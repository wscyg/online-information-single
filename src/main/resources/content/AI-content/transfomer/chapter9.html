    """Â∏¶Êé©Á†ÅÁöÑÂ§öÂ§¥Ê≥®ÊÑèÂäõÊú∫Âà∂"""

    def __init__(self, d_model, num_heads, dropout=0.1):
        super().__init__()
        assert d_model % num_heads == 0

        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads

        # Á∫øÊÄßÂèòÊç¢Â±Ç
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)

        self.dropout = nn.Dropout(dropout)
        self.scale = self.d_k ** -0.5

    def forward(self, query, key, value, mask=None):
        """
        Args:
            query: [batch_size, seq_len, d_model]
            key:   [batch_size, seq_len, d_model]
            value: [batch_size, seq_len, d_model]
            mask:  [batch_size, 1, seq_len, seq_len] or None

        Returns:
            output: [batch_size, seq_len, d_model]
            attention_weights: [batch_size, num_heads, seq_len, seq_len]
        """
        batch_size = query.size(0)

        # 1. Á∫øÊÄßÂèòÊç¢Âπ∂ÂàÜÂ§¥
        # [B, L, D] -> [B, L, H, D/H] -> [B, H, L, D/H]

        K = self.W_k(key).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        V = self.W_v(value).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)

        # 2. ËÆ°ÁÆóÊ≥®ÊÑèÂäõÂàÜÊï∞
        # [B, H, L, D/H] @ [B, H, D/H, L] -> [B, H, L, L]

        # 3. Â∫îÁî®Êé©Á†ÅÔºàÂ¶ÇÊûúÊúâÔºâ
        if mask is not None:
            # mask: [B, 1, L, L] -> ÂπøÊí≠Âà∞ [B, H, L, L]
            scores = scores.masked_fill(mask == 0, -1e9)

        # 4. Softmax
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout(attention_weights)

        # 5. Âä†ÊùÉÊ±ÇÂíå
        # [B, H, L, L] @ [B, H, L, D/H] -> [B, H, L, D/H]
        context = torch.matmul(attention_weights, V)

        # 6. ÊãºÊé•Â§öÂ§¥
        # [B, H, L, D/H] -> [B, L, H, D/H] -> [B, L, D]
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)

        # 7. ÊúÄÁªàÁ∫øÊÄßÂèòÊç¢
        output = self.W_o(context)

        return output, attention_weights
