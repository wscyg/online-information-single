import numpy as np
import matplotlib.pyplot as plt

class GradientDescent:
    def __init__(self, learning_rate=0.01, max_iter=1000):
        self.lr = learning_rate
        self.max_iter = max_iter
        self.costs = []

    def compute_cost(self, X, y, w, b, reg_type='none', lam=0):
        """ËÆ°ÁÆóÊçüÂ§±ÂáΩÊï∞"""
        m = X.shape[0]
        predictions = X.dot(w) + b
        cost = (1 / (2 * m)) * np.sum((predictions - y) ** 2)

        # Ê∑ªÂä†Ê≠£ÂàôÂåñÈ°π
        if reg_type == 'l1':
            cost += lam * np.sum(np.abs(w))
        elif reg_type == 'l2':
            cost += lam * np.sum(w ** 2)

        return cost

    def compute_gradients(self, X, y, w, b, reg_type='none', lam=0):
        """ËÆ°ÁÆóÊ¢ØÂ∫¶"""
        m = X.shape[0]
        predictions = X.dot(w) + b
        errors = predictions - y

        # Âü∫Á°ÄÊ¢ØÂ∫¶
        dw = (1 / m) * X.T.dot(errors)
        db = (1 / m) * np.sum(errors)

        # Ê∑ªÂä†Ê≠£ÂàôÂåñÊ¢ØÂ∫¶
        if reg_type == 'l1':
            dw += lam * np.sign(w)
        elif reg_type == 'l2':
            dw += 2 * lam * w

        return dw, db

    def fit(self, X, y, method='batch', batch_size=32,
             reg_type='none', lam=0):
        """ËÆ≠ÁªÉÊ®°Âûã"""
        m, n = X.shape
        self.w = np.random.normal(0, 0.01, n)
        self.b = 0
        self.costs = []

        for i in range(self.max_iter):
            if method == 'batch':
                # ÊâπÈáèÊ¢ØÂ∫¶‰∏ãÈôç
                dw, db = self.compute_gradients(X, y, self.w, self.b, reg_type, lam)

            elif method == 'stochastic':
                # ÈöèÊú∫Ê¢ØÂ∫¶‰∏ãÈôç
                idx = np.random.randint(0, m)
                dw, db = self.compute_gradients(X[idx:idx+1], y[idx:idx+1],
                                                 self.w, self.b, reg_type, lam)

            elif method == 'mini_batch':
                # Â∞èÊâπÈáèÊ¢ØÂ∫¶‰∏ãÈôç
                indices = np.random.choice(m, batch_size, replace=False)
                X_batch, y_batch = X[indices], y[indices]
                dw, db = self.compute_gradients(X_batch, y_batch, self.w, self.b,
                                                 reg_type, lam)

            # Êõ¥Êñ∞ÂèÇÊï∞
            self.w -= self.lr * dw
            self.b -= self.lr * db

            # ËÆ∞ÂΩïÊçüÂ§±
            if i % 10 == 0:
                cost = self.compute_cost(X, y, self.w, self.b, reg_type, lam)
                self.costs.append(cost)
                print(f"Ëø≠‰ª£ {i}, ÊçüÂ§±: {cost:.6f}")
